@startuml
actor Оператор as user
boundary "Веб-интерфейс" as ui
participant "Сервер" as server
database "База данных" as db
participant Устройство as device

activate user
user -> ui: Выбран пункт добавления устройства
activate ui
user <-- ui: Окно заполнения данных

user -> ui: Данные об устройстве (оборудовании)

ui -> server: Проверка на повторение устройства
activate server

server -> db: Запрос по устройству
activate db
db --> server: Ответ
deactivate db

alt Устройство уже добавлено

  server --> ui: Отказ в добавлении
  ui --> user: Сообщение об отказе

else Устройство ещё не добавлено

  server -> server: Сохранить устройство
  server -> db: Сохранить конфигурацию устройства
  activate db
  db -> db: Сохранение конфигурации
  db --> server: Конфигурация сохранена
  db--

  alt Устройство недоступно
    server -> device: Соединение
    ...Истечение длительности попытки подключения...
    server --> ui: Ошибка соединения с устройством
    ui --> user: Сообщение об ошибке
  else Устройство доступно
    server -> device: Соединение
    activate device
    device --> server: Ответ
    device--

    server --> ui: Добавление успешно
    server--
    ui --> user: Сообщение об успехе
    ui--
  end
end
user--
@enduml
